-- NOTE: THE TABLE SUMMARISES THE PORTAL INFORMATION WHICH ARE POSTED TO HUBSPOT CONTACTS
-- NOTE: DO NOT CHANGE THE COLUMN ORDER ETC. WITHOUT UPDATING DATA-LOADING

SELECT
	HUBSPOT_USER_ID
	, USER_ID
	, IFNULL(LAST_LOGIN, '2000-01-01') 			AS LAST_LOGIN
FROM {{ ref('core_dim_user') }} cdu
LEFT JOIN {{ ref('core_dim_company') }} cdc ON cdc.COMPANY_SK = cdu.OPERATIONAL_COMPANY_SK
WHERE HUBSPOT_USER_ID IS NOT NULL
	AND USER_ID IS NOT NULL
	-- NOTE: REMOVE KOSTAL USERS
	AND COMPANY_ID != 911300494
