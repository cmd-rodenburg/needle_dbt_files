SELECT C.ID												AS COMPANY_ID
	, C.NAME											AS COMPANY_NAME
	, C.STATUS											AS ACCOUNT_TYPE
	, C.SIGNEDCAPACITY									AS SIGNED_CAPACITY
	, IFF(C.CREATEDAT = '0001-01-01 00:00:00.0000000 +00:00'
		, NULL, C.CREATEDAT) 					::date	AS CREATE_DATE
	, C.LASTUPDATEDAT							::date 	AS LAST_UPDATED
	, TRY_TO_NUMBER(C.COMPANYCODE)						AS HUBSPOT_COMPANY_ID
FROM {{ source('operation', 'company') }} C
